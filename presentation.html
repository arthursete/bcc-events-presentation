Section 4 — Algorithms

SQL — Create Tables

Lookup Tables

CREATE TABLE Suburbs ( SuburbID INT PRIMARY KEY AUTO_INCREMENT, SuburbName VARCHAR(50) NOT NULL UNIQUE ); CREATE TABLE VenueTypes ( TypeID INT PRIMARY KEY AUTO_INCREMENT, TypeName VARCHAR(50) NOT NULL UNIQUE ); CREATE TABLE Users ( UserID INT PRIMARY KEY AUTO_INCREMENT, Username VARCHAR(50) NOT NULL UNIQUE, PasswordHash VARCHAR(255) NOT NULL, Role VARCHAR(20) DEFAULT 'admin', CreatedDate DATETIME DEFAULT CURRENT_TIMESTAMP );

Core Tables

CREATE TABLE Venues ( VenueID INT PRIMARY KEY AUTO_INCREMENT, VenueName VARCHAR(100) NOT NULL, VenueAddress VARCHAR(150) NOT NULL, Latitude DECIMAL(10,7) NOT NULL, Longitude DECIMAL(10,7) NOT NULL, SuburbID INT NOT NULL, FOREIGN KEY (SuburbID) REFERENCES Suburbs(SuburbID) ); CREATE TABLE VenueTypeLink ( VenueID INT NOT NULL, TypeID INT NOT NULL, PRIMARY KEY (VenueID, TypeID), FOREIGN KEY (VenueID) REFERENCES Venues(VenueID), FOREIGN KEY (TypeID) REFERENCES VenueTypes(TypeID) );

Section 4 — Algorithms

SQL — Select Queries

Venues in a Suburb (JOIN)

SELECT v.VenueName, v.VenueAddress, s.SuburbName FROM Venues v INNER JOIN Suburbs s ON v.SuburbID = s.SuburbID WHERE s.SuburbName = 'Chermside';

Search by Name (LIKE)

SELECT v.VenueName, s.SuburbName FROM Venues v INNER JOIN Suburbs s ON v.SuburbID = s.SuburbID WHERE v.VenueName LIKE '%Brigade%';

Filter by Type (Junction JOIN)

SELECT v.VenueName, s.SuburbName FROM Venues v INNER JOIN VenueTypeLink vtl ON v.VenueID = vtl.VenueID INNER JOIN VenueTypes vt ON vtl.TypeID = vt.TypeID INNER JOIN Suburbs s ON v.SuburbID = s.SuburbID WHERE vt.TypeName = 'Library';

Count per Suburb (GROUP BY)

SELECT s.SuburbName, COUNT(v.VenueID) AS VenueCount FROM Suburbs s LEFT JOIN Venues v ON s.SuburbID = v.SuburbID GROUP BY s.SuburbName ORDER BY VenueCount DESC;

Section 4 — Algorithms

Pseudocode — Search & Filter

Search Algorithm

FUNCTION SearchVenues(searchTerm) SET results TO empty list QUERY database: SELECT Venues, Suburbs, VenueTypes WHERE VenueName LIKE "%" + searchTerm + "%" OR SuburbName LIKE "%" + searchTerm + "%" OR TypeName LIKE "%" + searchTerm + "%" FOR EACH row in query results IF row.VenueID NOT already in results ADD row to results END IF END FOR RETURN results END FUNCTION

Filter Algorithm (Multi-Criteria)

FUNCTION FilterVenues(selectedSuburb, selectedType) SET baseQuery TO "SELECT ..." SET conditions TO empty list IF selectedSuburb is NOT empty THEN ADD "SuburbName = selectedSuburb" to conditions END IF IF selectedType is NOT empty THEN ADD "TypeName = selectedType" to conditions END IF IF conditions is NOT empty THEN APPEND " WHERE " + JOIN conditions WITH " AND " END IF EXECUTE baseQuery RETURN results END FUNCTION

Section 4 — HCI & Design

HCI & Design Principles

Applying CRAP principles and usability heuristics to ensure an intuitive interface.

Contrast

Key elements stand out through colour, size, and weight differences to guide the user's eye.

Primary action buttons use bold accent blue against the neutral background; headings are larger and heavier than body text.

Repetition

Consistent visual patterns across every page build familiarity and trust.

Same nav bar, card style, colour palette, and font on every page — users never feel lost.

Alignment

Every element is visually connected to something else — nothing is placed arbitrarily.

Search bar centred on Home; venue cards left-aligned in a grid; form labels aligned to their inputs.

Proximity

Related items are grouped together; unrelated items are separated by whitespace.

Filter controls grouped in a sidebar; venue name, address, and type displayed together on each card.

Accessibility & Usability

Designed for all users including those with disabilities or limited technical skills.

High-contrast text (WCAG AA), clear labels, keyboard navigation, descriptive alt text on images, and simple 3-click navigation paths.
